\begin{frame}{Problem Definition}
    \textbf{Goal:} Optimise personalised physical-activity interventions using RL.

    \vspace{4pt}
    \textbf{MDP:} \(\mathcal{M} = (\mathcal{S}, \mathcal{A}, \mathcal{P}, \mathcal{R}, \gamma)\)
    \begin{itemize}
        \item \(\mathcal{S}\): 36-state factored MDP (step bin, sleep, day of week, burden)
        \item \(\mathcal{A}\): 4 actions (idle, movement suggestion, goal reminder, journal)
        \item \(\mathcal{P}\): Unknown --- estimated via LLM simulation or hand-curated tables
        \item \(\mathcal{R}\): \(\alpha\!\cdot\!f(\text{step}) + (1{-}\alpha)\!\cdot\!g(\text{sleep}) - \lambda\!\cdot\!\mathbb{I}[\text{action} \neq \text{idle}]\)
    \end{itemize}

    \vspace{6pt}
    \textbf{Core challenge:} How do we iterate on RL agents without live patients?

    \vspace{4pt}
    \textit{Approach:} A config-driven simulation framework with LLM-bootstrapped user models.
\end{frame}

% --------------------------------------------------
% Slide 4 --- Our Approach (ethos \& novelty)
% --------------------------------------------------
\begin{frame}{Our Approach}
    \textbf{Two pillars:}

    \vspace{4pt}
    \begin{enumerate}
        \item \textbf{Config-Driven Framework}
        \begin{itemize}
            \item Entire MDP defined in YAML --- no source code changes between experiments
            \item Expression-based rewards, 3 transition types, 7 RL/bandit agents all registered by name
            \item A researcher can define state, actions, rewards, and transitions in one file
        \end{itemize}

        \item \textbf{Synthetic LLM Patients}
        \begin{itemize}
            \item LLMs as transition models --- sampled next-state probabilities from personas
            \item 5 behavioural archetypes: base, goal-driven, resistant, social responder, stable maintainer
            \item Plug in a new persona table, run the same experiment config
        \end{itemize}
    \end{enumerate}
