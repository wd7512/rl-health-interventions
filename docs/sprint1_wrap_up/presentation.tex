\begin{frame}{Problem Definition}
    \textbf{Goal:} Optimise personalised physical-activity interventions using RL.

    \vspace{4pt}
    \textbf{MDP formulation:} \(\mathcal{M} = (\mathcal{S}, \mathcal{A}, \mathcal{P}, \mathcal{R}, \gamma)\)
    \begin{itemize}
        \item \(\mathcal{S}\): 36-state factored MDP (step bin, sleep quality, day of week, burden)
        \item \(\mathcal{A}\): 4 actions (idle, movement suggestion, goal reminder, journal)
        \item \(\mathcal{P}\): Unknown --- estimated via LLM simulation or hand-curated tables
        \item \(\mathcal{R}\): \(\alpha\!\cdot\!f(\text{step\_bin}) + (1{-}\alpha)\!\cdot\!g(\text{sleep}) - \lambda\!\cdot\!\mathbb{I}[\text{action} \neq \text{idle}]\)
    \end{itemize}

    \vspace{6pt}
    \textbf{Core challenge:} How do we iterate on RL agents without live patients?

    \vspace{4pt}
    \textit{Approach:} Build a config-driven simulation framework with LLM-bootstrapped user models.
\end{frame}

% --------------------------------------------------
% Slide 4 --- Our Approach
% --------------------------------------------------
\begin{frame}{Our Approach}
    \textbf{Two pillars:}

    \vspace{4pt}
    \begin{enumerate}
        \item \textbf{Config-Driven Framework}
        \begin{itemize}
            \item Entire MDP defined in YAML --- no source code changes
            \item 36-state factored environment, expression-based rewards, 3 transition types
            \item 7 RL/bandit agents, evaluation runner with multi-seed benchmarking
        \end{itemize}

        \item \textbf{Synthetic LLM Patients}
        \begin{itemize}
            \item 22{,}320 prompts per persona, 5 distinct behavioural personas
            \item LLM-as-transition-model: DeepSeek V4 Flash samples next-state probabilities
            \item Total API cost: \textbf{\$0.27} for the entire bootstrap
        \end{itemize}
    \end{enumerate}

\begin{frame}{What is Next}
    \begin{enumerate}
        \item \textbf{Sensor Foundation Models}
        \begin{itemize}
            \item Wearable FM POC (NormWear, 768-dim embeddings) --- feature branch
            \item FM-guided dynamic action spaces (15 interventions, 3 archetypes)
        \end{itemize}

        \item \textbf{Additional LLM Backends}
        \begin{itemize}
            \item Beyond DeepSeek V4 Flash: GLM 5.2, Qwen, Claude
            \item Multi-turn persona dialogues for richer state transitions
        \end{itemize}

        \item \textbf{Prompt Engineering}
        \begin{itemize}
            \item COM-B behaviour-change taxonomy integration
            \item Structured JSON schema enforcement, chain-of-thought prompting
        \end{itemize}

        \item \textbf{Synthetic Clinical Results}
        \begin{itemize}
            \item Calibrate transition tables to real-world effect sizes (PEARL: +200--300 steps)
            \item Integration with wearable datasets (All of Us, UK Biobank)
        \end{itemize}
    \end{enumerate}
\end{frame}
