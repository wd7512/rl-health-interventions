\begin{frame}{Problem Definition}
    \textbf{Goal:} Optimise personalised physical-activity interventions with reinforcement learning.

    \vspace{4pt}
    \textbf{MDP formulation:} \(\mathcal{M} = (\mathcal{S}, \mathcal{A}, \mathcal{P}, \mathcal{R}, \gamma)\)
    \begin{itemize}
        \item \(\mathcal{S}\): 36-state factored MDP (step bin, sleep quality, day of week, burden)
        \item \(\mathcal{A}\): 4 actions (idle, movement suggestion, goal reminder, journal)
        \item \(\mathcal{P}\): Unknown transition dynamics, estimated from LLM simulation or curated tables
        \item \(\mathcal{R}\): \(\alpha\!\cdot\!f(\text{step\_bin}) + (1{-}\alpha)\!\cdot\!g(\text{sleep}) - \lambda\!\cdot\!\mathbb{I}[\text{action} \neq \text{idle}]\)
    \end{itemize}

    \vspace{6pt}
    \textbf{Core challenge:} How do we iterate on RL agents without live patients?

    \vspace{4pt}
    \textit{Approach:} Use a config-driven simulator with LLM-bootstrapped user models.
\end{frame}

% --------------------------------------------------
% Slide 4 --- Our Approach (ethos \& novelty)
% --------------------------------------------------
\begin{frame}{Our Approach}
    \textbf{Two pillars:}

    \vspace{4pt}
    \begin{enumerate}
        \item \textbf{Config-Driven Framework}
        \begin{itemize}
            \item Entire MDP defined in YAML, with no source-code changes
            \item 36-state factored environment, expression-based rewards, 3 transition types
            \item 7 RL/bandit agents, evaluation runner with multi-seed benchmarking
        \end{itemize}

        \item \textbf{Synthetic LLM Patients}
        \begin{itemize}
            \item 22{,}320 prompts per persona across 5 behavioural personas
            \item LLM-as-transition-model samples next-state probabilities
            \item Total API cost: \textbf{\$0.27} for the entire bootstrap
        \end{itemize}
    \end{enumerate}

    \vspace{6pt}

% Slide 7 --- Next Steps

\begin{frame}{Next Steps}
    \small
    \begin{enumerate}
        \item \textbf{Broader Agent Suite}
        \begin{itemize}
            \item Add PEARL-style contextual bandit/RL agents for direct comparison
            \item Stress-test offline RL, constrained RL, and adaptive bandit baselines
        \end{itemize}

        \item \textbf{Deep Learning Agents}
        \begin{itemize}
            \item Benchmark DQN, actor--critic, and sequence-based policies
            \item Explore representation learning for high-dimensional wearable state inputs
        \end{itemize}

        \item \textbf{More Complex Environments}
        \begin{itemize}
            \item Expand state spaces with context, history, fatigue, and adherence dynamics
            \item Model non-stationarity, delayed rewards, and intervention burden
        \end{itemize}

        \item \textbf{Sensor Foundation Models}
        \begin{itemize}
            \item Wearable foundation-model embeddings for dynamic action spaces
            \item Connect interventions to sensor-derived activity and sleep context
        \end{itemize}

        \item \textbf{LLM and Prompt Variants}
        \begin{itemize}
            \item Compare DeepSeek, GLM, Qwen, Claude, and smaller local models
            \item Test COM-B prompts, structured JSON, and multi-turn persona dialogues
        \end{itemize}
    \end{enumerate}
\end{frame}
